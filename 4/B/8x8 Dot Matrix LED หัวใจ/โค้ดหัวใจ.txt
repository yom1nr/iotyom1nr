#include <MD_MAX72xx.h>
#include <SPI.h>


#define HARDWARE_TYPE MD_MAX72XX::FC16_HW
#define MAX_DEVICES 1
#define CS_PIN 5


MD_MAX72XX mx = MD_MAX72XX(HARDWARE_TYPE, CS_PIN, MAX_DEVICES);




uint8_t heart[8] = {
  0b00000000,
  0b01100110,
  0b11111111,
  0b11111111,
  0b11111111,
  0b01111110,
  0b00111100,
  0b00011000
};


void setup() {
  mx.begin();
  mx.control(MD_MAX72XX::INTENSITY, 6); // ความสว่าง 0–15
  mx.clear();


  for (int row = 0; row < 8; row++) {
    mx.setRow(0, row, heart[row]);
  }
}


void loop() {
}
